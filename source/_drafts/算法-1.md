# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class TreePrinter:
    def printTree(self, root):
        # write code here
        q = []
        li = []
        temp = []
        last = root
        nlast = root
        q.append(root)
        while(len(q)&gt;0):
            n = q.pop(0)
            temp.append(n.val)
            if(n.left):
                q.append(n.left)
                nlast = n.left
            if(n.right):
                q.append(n.right)
                nlast = n.right
            if(last==n):
                li.append(temp)
                temp = []
                last = nlast
        return li</pre>
